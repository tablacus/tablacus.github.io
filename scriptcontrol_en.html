  • Script Control for 64-bit platforms.
  • Alternate MSScript.ocx
* Install & Uninstall

Run setup.exe

How to use (Same as Microsoft Script Control)

ex1.js 
var SC = new ActiveXObject("ScriptControl");
SC.Language = "VBScript";
SC.ExecuteStatement('MsgBox 1+2');
ex2.js 
var SC = new ActiveXObject("ScriptControl");
SC.Language = "VBScript";
WScript.Echo(SC.Eval("1+2"));
ex3.js 
var SC = new ActiveXObject("ScriptControl");
SC.Language = "JScript";
SC.AddObject("WScript", WScript);
SC.AddCode('function fnx(a,b,c){ WScript.Echo(a + ":" + b + ":" + c) }');
SC.Run("fnx", 1, "data2", "data3");
ex4.js 
var SC = new ActiveXObject("ScriptControl");
SC.Language = "JScript";
SC.AddObject("WScript", WScript);
SC.AddCode('function fnx(a,b,c){ WScript.Echo(a + ":" + b + ":" + c) }');
SC.CodeObject.fnx(1,2,3);
ex5.vbs 
Set SC = CreateObject("ScriptControl")
SC.Language = "VBScript"
MsgBox SC.Eval("1+2")
